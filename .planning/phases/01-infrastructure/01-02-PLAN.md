2. Define schema.prisma with all 6 core models:

```prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

model User {
  id                 String           @id @default(uuid())
  openId             String           @unique @map("open_id")
  nickname           String?
  avatar             String?
  subscriptionTier   SubscriptionTier @default(free) @map("subscription_tier")
  subscriptionEndsAt DateTime?        @map("subscription_ends_at")
  createdAt          DateTime         @default(now()) @map("created_at")
  contacts           Contact[]
  interactions       Interaction[]
  relationshipScores RelationshipScore[]
  reminders          Reminder[]
  businessCards      BusinessCard[]

  @@map("users")
}

enum SubscriptionTier {
  free
  premium
}

model Contact {
  id           String     @id @default(uuid())
  userId       String     @map("user_id")
  name         String
  company      String?
  title        String?
  phone        String?
  wechatId     String?    @map("wechat_id")
  email        String?
  avatar       String?
  source       String?
  tags         String[]
  customFields Json?      @map("custom_fields")
  createdAt    DateTime   @default(now()) @map("created_at")
  updatedAt    DateTime   @updatedAt @map("updated_at")
  user         User       @relation(fields: [userId], references: [id])
  interactions Interaction[]
  reminders    Reminder[]
  businessCards BusinessCard[]

  @@index([userId])
  @@index([name])
  @@index([tags])
  @@map("contacts")
}

model Interaction {
  id          String   @id @default(uuid())
  contactId   String   @map("contact_id")
  userId      String   @map("user_id")
  type        String   // note, call, meeting, message, etc.
  content     String
  duration    Int?     // duration in seconds, null for notes
  occurredAt  DateTime @map("occurred_at")
  createdAt   DateTime @default(now()) @map("created_at")
  contact     Contact  @relation(fields: [contactId], references: [id])
  user        User     @relation(fields: [userId], references: [id])

  @@index([contactId])
  @@index([userId])
  @@index([occurredAt])
  @@map("interactions")
}

model RelationshipScore {
  id                   String   @id @default(uuid())
  contactId            String   @map("contact_id")
  userId               String   @map("user_id")
  score                Float
  lastContactDays      Int      @map("last_contact_days")
  interactionFrequency Float    @map("interaction_frequency")
  emotionalTone        String?  @map("emotional_tone")
  calculatedAt         DateTime @map("calculated_at")
  contact              Contact  @relation(fields: [contactId], references: [id])
  user                 User     @relation(fields: [userId], references: [id])

  @@index([contactId])
  @@index([userId])
  @@map("relationship_scores")
}

model Reminder {
  id            String   @id @default(uuid())
  userId        String   @map("user_id")
  contactId     String?  @map("contact_id")
  type          String   // relationship, birthday, custom
  message       String
  scheduledAt   DateTime @map("scheduled_at")
  sentAt        DateTime? @map("sent_at")
  recurrenceRule String? @map("recurrence_rule")
  user          User     @relation(fields: [userId], references: [id])
  contact       Contact? @relation(fields: [contactId], references: [id])

  @@index([userId])
  @@index([scheduledAt])
  @@map("reminders")
}

model BusinessCard {
  id        String   @id @default(uuid())
  userId    String   @map("user_id")
  contactId String?  @map("contact_id")
  imageUrl  String   @map("image_url")
  ocrData   Json?    @map("ocr_data")
  createdAt DateTime @default(now()) @map("created_at")
  user      User     @relation(fields: [userId], references: [id])
  contact   Contact? @relation(fields: [contactId], references: [id])

  @@index([userId])
  @@map("business_cards")
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Backend -> PostgreSQL | Database credentials in env vars; connection string could be logged |
| Backend -> Redis | Cache layer; Redis has no auth in dev mode |
| docker-compose -> Host | Port mappings expose services to localhost |
| Shared types -> Frontend/Backend | Type drift between Prisma schema and shared types |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-05 | Information Disclosure | Database credentials | mitigate | DATABASE_URL in .env only, never committed; docker-compose uses .env.docker with dev-only credentials; production uses secrets manager |
| T-01-06 | Information Disclosure | Redis (no auth in dev) | accept | Dev-only Redis has no auth; production Redis requires password via REDIS_URL with auth component; acceptable risk for local development |
| T-01-07 | Tampering | Prisma schema vs shared types drift | mitigate | Shared types are manually kept in sync with Prisma schema; add automated type-check comparison in CI (Phase 8) |
| T-01-08 | Denial of Service | Database connection pool | mitigate | PrismaClient singleton pattern in src/db/client.ts prevents pool exhaustion on dev hot reload; production will add connection pool limits |
| T-01-09 | Information Disclosure | Docker dev credentials | mitigate | .env.docker contains dev-only passwords (changlianxi_dev); production Docker uses Docker secrets or mounted env from secrets manager |
</threat_model>
