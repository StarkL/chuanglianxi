       Step 3: Create `frontend/scripts/generate-tabbar-icons.mjs`:
       ```javascript
       import sharp from 'sharp';
       import { readFileSync, writeFileSync } from 'fs';
       import { join, dirname } from 'path';
       import { fileURLToPath } from 'url';

       const __dirname = dirname(fileURLToPath(import.meta.url));
       const tabbarDir = join(__dirname, '../src/static/tabbar/');

       const icons = [
         { svg: 'contacts.svg', png: 'contacts.png' },
         { svg: 'contacts-active.svg', png: 'contacts-active.png' },
         { svg: 'reminders.svg', png: 'reminders.png' },
         { svg: 'reminders-active.svg', png: 'reminders-active.png' },
         { svg: 'mine.svg', png: 'mine.png' },
         { svg: 'mine-active.svg', png: 'mine-active.png' },
       ];

       for (const { svg, png } of icons) {
         const svgPath = join(tabbarDir, svg);
         const pngPath = join(tabbarDir, png);
         const svgContent = readFileSync(svgPath, 'utf-8');
         await sharp(Buffer.from(svgContent))
           .resize(81, 81, { fit: 'contain', background: { r: 0, g: 0, b: 0, alpha: 0 } })
           .png()
           .toFile(pngPath);
         console.log(`Generated ${png}`);
       }
       ```
