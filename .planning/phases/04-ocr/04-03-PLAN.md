    Step C -- Frontend Page: Create card wall
    1. Create frontend/src/pages/ocr/wall/wall.vue:

    Script:
    ```typescript
    import { ref, onMounted } from 'vue'
    import { getBusinessCards, type BusinessCardItem } from '../../../api/ocr.js'

    const cards = ref<BusinessCardItem[]>([])
    const loading = ref(true)

    function extractName(ocrData: Record<string, unknown> | null): string {
      if (!ocrData) return '未命名'
      return (ocrData.name as string) || '未命名'
    }

    function extractCompany(ocrData: Record<string, unknown> | null): string {
      if (!ocrData) return ''
      return (ocrData.company as string) || ''
    }

    function formatDate(dateStr: string): string {
      const d = new Date(dateStr)
      return `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, '0')}-${String(d.getDate()).padStart(2, '0')}`
    }

    function handleCardTap(card: BusinessCardItem) {
      uni.navigateTo({ url: `/pages/ocr/result/result?cardId=${card.id}` })
    }

    onMounted(async () => {
      try {
        const res = await getBusinessCards()
        if (res.success && res.data) {
          cards.value = res.data
        }
      } catch {
        uni.showToast({ title: '加载失败', icon: 'none' })
      } finally {
        loading.value = false
      }
    })
    ```

    Template:
    ```vue
    <view class="wall-page">
      <view v-if="loading" class="loading">
        <text class="loading-text">加载中...</text>
      </view>
      <view v-else-if="cards.length === 0" class="empty">
        <text class="empty-title">暂无名片</text>
        <text class="empty-hint">拍摄名片即可添加</text>
      </view>
      <view v-else class="wall">
        <view v-for="card in cards" :key="card.id" class="card" @click="handleCardTap(card)">
          <image class="card-image" :src="card.imageUrl" mode="aspectFill" />
          <view class="card-info">
            <text class="card-company" v-if="extractCompany(card.ocrData)">
              {{ extractCompany(card.ocrData) }}
            </text>
            <text class="card-name">{{ extractName(card.ocrData) }}</text>
            <text class="card-date">{{ formatDate(card.createdAt) }}</text>
          </view>
        </view>
      </view>
    </view>
    ```
